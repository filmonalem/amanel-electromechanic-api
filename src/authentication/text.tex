import { Injectable, NotFoundException, BadRequestException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Allocation } from './entities/allocation.entity';
import { Client } from '../client/entities/client.entity';
import { Property } from '../properties/entities/property.entity';
import { validate as isValidUUID } from 'uuid';
import { v4 as uuidv4 } from 'uuid';

@Injectable()
export class AllocationService {
  constructor(
    @InjectRepository(Allocation)
    private allocationRepository: Repository<Allocation>,
    @InjectRepository(Property)
    private propertyRepository: Repository<Property>,
    @InjectRepository(Client)
    private clientRepository: Repository<Client>,
  ) {}

  /**
   * Creates multiple random allocations
   * @param count Number of allocations to create
   * @returns Array of created allocations
   */
  async createRandomAllocations(count: number): Promise<Allocation[]> {
    if (!Number.isInteger(count) || count < 1) {
      throw new BadRequestException('Count must be a positive integer');
    }

    const queryRunner = this.allocationRepository.manager.connection.createQueryRunner();
    await queryRunner.connect();
    await queryRunner.startTransaction();

    try {
      // Fetch all active clients in one go
      const clients = await queryRunner.manager
        .createQueryBuilder(Client, 'client')
        .setLock('pessimistic_write')
        .where('client.isActive = :isActive', { isActive: true })
        .orderBy('RANDOM()')
        .limit(count)
        .getMany();

      if (clients.length === 0) {
        throw new NotFoundException('No active clients found');
      }
      if (clients.length < count) {
        throw new BadRequestException(`Only ${clients.length} active clients available, requested ${count}`);
      }

      // Fetch properties
      const properties = await queryRunner.manager
        .createQueryBuilder(Property, 'property')
        .setLock('pessimistic_write')
        .orderBy('RANDOM()')
        .limit(count)
        .getMany();

      if (properties.length === 0) {
        throw new NotFoundException('No properties found');
      }
      if (properties.length < count) {
        throw new BadRequestException(`Only ${properties.length} properties available, requested ${count}`);
      }

      const allocations: Allocation[] = [];
      const propertyRoomMap = new Map<string, number[]>();

      // Initialize available rooms for each property
      for (const property of properties) {
        const allocatedRooms = await queryRunner.manager
          .createQueryBuilder(Allocation, 'allocation')
          .select('allocation.roomNo')
          .where('allocation.propertyId = :propertyId', { propertyId: property.propertyId })
          .getRawMany();

        const allocatedRoomNumbers = allocatedRooms.map((alloc) => alloc.roomNo);
        const availableRooms = Array.from({ length: property.numberOfRooms }, (_, i) => i + 1).filter(
          (room) => !allocatedRoomNumbers.includes(room),
        );

        if (availableRooms.length === 0) {
          throw new BadRequestException(`No available rooms for property ${property.propertyId}`);
        }

        propertyRoomMap.set(property.propertyId, availableRooms);
      }

      // Create allocations
      for (let i = 0; i < count; i++) {
        const client = clients[i];
        const property = properties[i];

        const availableRooms = propertyRoomMap.get(property.propertyId);
        if (!availableRooms || availableRooms.length === 0) {
          throw new BadRequestException(`No available rooms for property ${property.propertyId}`);
        }

        const roomNo = availableRooms.shift(); // Take the first available room
        const allocationId = uuidv4();

        if (!isValidUUID(allocationId)) {
          throw new BadRequestException('Invalid UUID generated');
        }

        const allocation = this.allocationRepository.create({
          allocationId,
          propertyId: property.propertyId,
          customerId: client.clientId,
          roomNo,
          address: property.address,
          block: property.block,
          note: `Random allocation for ${client.fullName}`,
          createAt: new Date(),
        });

        allocations.push(allocation);
      }

      // Save all allocations
      const savedAllocations = await queryRunner.manager.save(Allocation, allocations);
      await queryRunner.commitTransaction();
      return savedAllocations;
    } catch (error) {
      await queryRunner.rollbackTransaction();
      throw error;
    } finally {
      await queryRunner.release();
    }
  }
}
